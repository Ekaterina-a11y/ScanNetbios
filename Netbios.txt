#!/usr/bin/env python3
"""
NETBIOS SCANNER - Scan du port 139 (TOUTES les IPs)
Usage: python3 netbios_scan.py <fichier_ips.txt>
"""

import socket
import sys
import ipaddress
import subprocess
import threading
import time
import os

try:
    from colorama import init, Fore, Style
    init(autoreset=True)
except:
    class Fore: RED=CYAN=GREEN=YELLOW=BLUE=MAGENTA=WHITE=BLACK=''
    Style = type('Style', (), {'RESET_ALL': ''})()

# Variables globales
total_ips = 0
scanned_ips = 0
found_ips = 0
lock = threading.Lock()
results = []
start_time = 0
running = True

def print_banner():
    print(f"""
{Fore.CYAN}╔{'═' * 58}╗
║{Fore.WHITE}  NETBIOS SCANNER - Port 139 (SMB/NetBIOS){Fore.CYAN}            ║
║{Fore.WHITE}  Scan de TOUTES les IPs (publiques et privées){Fore.CYAN}       ║
╚{'═' * 58}╝{Style.RESET_ALL}
""")

def scan_port(ip):
    """Test si le port 139 est ouvert"""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(0.5)
        result = sock.connect_ex((ip, 139))
        sock.close()
        return result == 0
    except:
        return False

def get_hostname(ip):
    try:
        return socket.gethostbyaddr(ip)[0]
    except:
        return "N/A"

def get_mac(ip):
    """MAC seulement pour les IPs locales"""
    try:
        if ip.startswith(('192.168.', '10.', '172.16.', '172.17.', '172.18.', '172.19.', '172.20.', '172.21.', '172.22.', '172.23.', '172.24.', '172.25.', '172.26.', '172.27.', '172.28.', '172.29.', '172.30.', '172.31.')):
            if sys.platform.startswith('linux'):
                result = subprocess.run(['arp', '-n', ip], capture_output=True, text=True, timeout=0.5)
                for line in result.stdout.split('\n'):
                    if ip in line:
                        parts = line.split()
                        if len(parts) >= 3:
                            return parts[2]
    except:
        pass
    return "N/A (publique)"

def get_vendor(mac):
    if mac == "N/A" or mac == "N/A (publique)":
        return "N/A"
    
    oui = {
        '00:00:0C': 'Cisco', '00:0C:29': 'VMware', '00:50:56': 'VMware',
        '00:05:69': 'HP', '00:1A:A0': 'HP', '00:11:22': 'Apple',
        '00:16:CB': 'Apple', '00:1E:C2': 'Apple', '00:25:00': 'Apple',
        '00:E0:4C': 'Realtek', '00:1A:8C': 'Dell', '00:1D:09': 'Dell',
        '00:1F:29': 'Dell', '00:23:AE': 'Samsung', '00:0F:EA': 'Broadcom',
        '00:10:18': 'Microsoft', '00:15:5D': 'Microsoft',
    }
    
    for prefix, name in oui.items():
        if mac.upper().startswith(prefix):
            return name
    return "Inconnu"

def get_netbios(ip):
    try:
        result = subprocess.run(['nmblookup', '-A', ip], capture_output=True, text=True, timeout=1)
        for line in result.stdout.split('\n'):
            if '<00>' in line and 'GROUP' not in line:
                parts = line.split()
                if parts:
                    return parts[0]
    except:
        pass
    return "N/A"

def save_result(ip, hostname, netbios, mac, vendor):
    try:
        with open("resultats_netbios.txt", "a") as f:
            f.write(f"IP: {ip}\n")
            f.write(f"Hostname: {hostname}\n")
            f.write(f"NetBIOS: {netbios}\n")
            f.write(f"MAC: {mac}\n")
            f.write(f"Fabricant: {vendor}\n")
            f.write("=" * 40 + "\n\n")
    except:
        pass

def print_result(ip, hostname, netbios, mac, vendor):
    print(f"\n{Fore.GREEN}{'=' * 60}")
    print(f"{Fore.GREEN}[+] PORT 139 OUVERT !")
    print(f"{Fore.GREEN}{'=' * 60}")
    print(f"{Fore.WHITE}  IP       : {Fore.CYAN}{ip}")
    print(f"{Fore.WHITE}  Hostname : {Fore.YELLOW}{hostname}")
    print(f"{Fore.WHITE}  NetBIOS  : {Fore.MAGENTA}{netbios}")
    print(f"{Fore.WHITE}  MAC      : {Fore.BLUE}{mac}")
    print(f"{Fore.WHITE}  Fabricant: {Fore.GREEN}{vendor}")
    print(f"{Fore.GREEN}{'=' * 60}{Style.RESET_ALL}")

def display_progress():
    """Affiche la progression"""
    global scanned_ips, total_ips, found_ips, start_time
    
    with lock:
        elapsed = time.time() - start_time
        percent = (scanned_ips / total_ips * 100) if total_ips > 0 else 0
        speed = scanned_ips / elapsed if elapsed > 0 else 0
        
        # Barre de progression
        bar_len = 40
        filled = int(bar_len * scanned_ips / total_ips) if total_ips > 0 else 0
        bar = f"{Fore.GREEN}{'█' * filled}{Fore.WHITE}{'░' * (bar_len - filled)}{Style.RESET_ALL}"
        
        # Calcul du temps restant
        if speed > 0:
            remaining = (total_ips - scanned_ips) / speed
            if remaining > 3600:
                eta = f"{int(remaining//3600)}h {int((remaining%3600)//60)}m"
            elif remaining > 60:
                eta = f"{int(remaining//60)}m {int(remaining%60)}s"
            else:
                eta = f"{int(remaining)}s"
        else:
            eta = "calcul..."
        
        sys.stdout.write('\r')
        sys.stdout.write(f"{Fore.CYAN}[{bar}] {percent:.1f}% | "
                        f"{scanned_ips}/{total_ips} | "
                        f"Ouverts: {Fore.GREEN}{found_ips}{Fore.CYAN} | "
                        f"Vitesse: {speed:.1f} ips/s | "
                        f"ETA: {eta}{Style.RESET_ALL}")
        sys.stdout.flush()

def scan_ip(ip):
    """Scanne une seule IP - TOUTES les IPs sont scannées"""
    global scanned_ips, found_ips
    
    try:
        # Scan du port 139 (TOUTES les IPs, pas de filtre)
        if scan_port(ip):
            hostname = get_hostname(ip)
            mac = get_mac(ip)
            vendor = get_vendor(mac)
            netbios = get_netbios(ip)
            
            with lock:
                found_ips += 1
            
            print_result(ip, hostname, netbios, mac, vendor)
            save_result(ip, hostname, netbios, mac, vendor)
        
        with lock:
            scanned_ips += 1
            
    except Exception as e:
        with lock:
            scanned_ips += 1

def worker(ip_list, start_idx, end_idx):
    """Thread worker"""
    for i in range(start_idx, end_idx):
        if not running:
            break
        scan_ip(ip_list[i])
        display_progress()

def main():
    global total_ips, start_time, running
    
    print_banner()
    
    if len(sys.argv) != 2:
        print(f"{Fore.RED}Usage: python3 {sys.argv[0]} <fichier_ips.txt>")
        print(f"{Fore.YELLOW}Exemple: python3 {sys.argv[0]} ips.txt{Style.RESET_ALL}")
        sys.exit(1)
    
    input_file = sys.argv[1]
    
    # Charger les IPs
    ip_list = []
    try:
        with open(input_file, 'r') as f:
            for line in f:
                line = line.strip()
                if line and not line.startswith('#'):
                    try:
                        if '/' in line:
                            net = ipaddress.ip_network(line, strict=False)
                            for ip in net.hosts():
                                ip_list.append(str(ip))
                        else:
                            ip_list.append(line)
                    except:
                        continue
    except FileNotFoundError:
        print(f"{Fore.RED}[!] Fichier introuvable: {input_file}{Style.RESET_ALL}")
        sys.exit(1)
    
    total_ips = len(ip_list)
    
    if total_ips == 0:
        print(f"{Fore.RED}[!] Aucune IP valide trouvée{Style.RESET_ALL}")
        sys.exit(1)
    
    print(f"{Fore.CYAN}[*] Fichier chargé: {total_ips:,} IPs à scanner")
    print(f"{Fore.CYAN}[*] Scan du port 139 (NetBIOS/SMB)")
    print(f"{Fore.YELLOW}[*] TOUTES les IPs seront scannées (publiques et privées)")
    print(f"{Fore.YELLOW}[*] Appuyez sur Ctrl+C pour arrêter{Style.RESET_ALL}\n")
    
    # Supprimer ancien fichier
    if os.path.exists("resultats_netbios.txt"):
        os.remove("resultats_netbios.txt")
    
    start_time = time.time()
    
    # Diviser le travail entre threads
    num_threads = min(200, total_ips)  # Augmenté à 200 threads
    chunk_size = total_ips // num_threads
    
    threads = []
    
    for i in range(num_threads):
        start_idx = i * chunk_size
        end_idx = start_idx + chunk_size if i < num_threads - 1 else total_ips
        t = threading.Thread(target=worker, args=(ip_list, start_idx, end_idx))
        t.daemon = True
        t.start()
        threads.append(t)
    
    # Attendre la fin
    try:
        for t in threads:
            t.join()
    except KeyboardInterrupt:
        running = False
        print(f"\n\n{Fore.YELLOW}[!] Scan interrompu par l'utilisateur{Style.RESET_ALL}")
    
    # Afficher le résumé
    elapsed = time.time() - start_time
    print(f"\n\n{Fore.GREEN}{'=' * 60}")
    print(f"{Fore.GREEN}           SCAN TERMINÉ")
    print(f"{Fore.GREEN}{'=' * 60}")
    print(f"{Fore.WHITE}  IPs scannées    : {Fore.CYAN}{scanned_ips:,}/{total_ips:,}")
    print(f"{Fore.WHITE}  Ports 139 ouverts : {Fore.GREEN}{found_ips}")
    print(f"{Fore.WHITE}  Temps total      : {Fore.YELLOW}{elapsed:.1f} secondes ({elapsed/60:.1f} minutes)")
    if elapsed > 0:
        print(f"{Fore.WHITE}  Vitesse moyenne  : {Fore.CYAN}{scanned_ips/elapsed:.1f} ips/s")
    print(f"{Fore.WHITE}  Résultats        : {Fore.MAGENTA}resultats_netbios.txt")
    print(f"{Fore.GREEN}{'=' * 60}{Style.RESET_ALL}")

if __name__ == "__main__":
    try:
        main()
    except KeyboardInterrupt:
        print(f"\n{Fore.YELLOW}[!] Interrompu{Style.RESET_ALL}")
        sys.exit(0)
