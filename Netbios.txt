import socket

try:
    from colorama import init, Fore, Back, Style
    init(autoreset=True, strip=False)  # strip=False pour proxychains
except ImportError:
    class Fore: BLACK = RED = GREEN = YELLOW = BLUE = MAGENTA = CYAN = WHITE = '' 
    class Back: BLACK = RED = GREEN = YELLOW = BLUE = MAGENTA = CYAN = WHITE = ''
    Style = type('Style', (), {'RESET_ALL': '', 'DIM': '', 'BRIGHT': ''})()

class NetbiosProScanner:
    def __init__(self):
        self.total_ips = 0
        self.scanned_ips = 0
        self.found_ips = 0
        self.lock = threading.Lock()
        self.results = []
        self.start_time = 0
        self.running = True
        self.banner_printed = False
        self.progress_thread = None
        self.executor = None
        self.futures = []
        self.proxychains_mode = False
        
        # Détection proxychains
        if 'proxychains' in ' '.join(sys.argv).lower():
            self.proxychains_mode = True
            print(f"{Fore.YELLOW}🔗 Mode PROXYCHAINS détecté - Optimisations activées{Style.RESET_ALL}")
        
        # Signal handler
        signal.signal(signal.SIGINT, self.signal_handler)
        atexit.register(self.cleanup)

    def signal_handler(self, signum, frame):
        print(f"\n{Fore.YELLOW}⏹️  Ctrl+C détecté ! Arrêt propre...{Style.RESET_ALL}")
        self.running = False
        self.save_all_results()
        self.cleanup()
        os._exit(0)  # Force exit même avec proxychains

    def cleanup(self):
        self.running = False
        if self.executor:
            self.executor.shutdown(wait=False, cancel_futures=True)

    def safe_print(self, text):
        """Print non-bloquant compatible proxychains"""
        try:
            sys.stdout.write(text)
            sys.stdout.flush()
        except:
            pass

    def print_banner(self):
        if self.proxychains_mode:
            banner = f"""
{Fore.CYAN}

  ╔══════════════════════════════════════╗

  ║ ⚡ Scan via proxychains4              ║
  ╚══════════════════════════════════════╝
{Fore.RESET_ALL}
            """
        else:
            banner = f"""
{Fore.CYAN}{Style.BRIGHT}
    ██████╗██╗  ██╗██████╗  ██████╗██╗  ██╗
   ██╔════╝██║  ██║██╔══██╗██╔═══██╗██║ ██╔╝
   ██║     ███████║██████╔╝██║   ██║█████╔╝ 
   ██║     ██╔══██║██╔══██╗██║   ██║██╔═██╗ 
   ╚██████╗██║  ██║██████╔╝╚██████╔╝██║  ██╗
    ╚═════╝╚═╝  ╚═╝╚═════╝  ╚═════╝ ╚═╝  ╚═╝
{Style.RESET_ALL}
{Fore.GREEN}v2.2 - PROXYCHAINS 100% Compatible{Style.RESET_ALL}
            """
        self.safe_print(banner + "\n")

    def scan_port_proxychains(self, ip, ports=[139, 445]):
        """Scan compatible proxychains - timeout très court"""
        for port in ports:
            if not self.running:
                return None
            try:
                sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
                sock.settimeout(1.2)  # Timeout plus long pour proxy
                result = sock.connect_ex((ip, port))
                sock.close()
                if result == 0:
                    return port
                time.sleep(0.01)  # Pause anti-flood proxy
            except:
                time.sleep(0.02)  # Retry proxy
                continue
        return None

    def get_netbios_proxy_safe(self, ip):
        """NetBIOS sans subprocess (problématique avec proxychains)"""
        return "NetBIOS-Désactivé" if self.proxychains_mode else self.get_netbios_name(ip)

    def get_netbios_name(self, ip):
        """Version normale NetBIOS"""
        if self.proxychains_mode:
            return "N/A (proxy)"
        try:
            result = subprocess.run(['nmblookup', '-A', ip], capture_output=True, 
                                  text=True, timeout=1.5)
            match = re.search(r'([A-Z0-9\-]{1,15})<00>', result.stdout)
            return match.group(1) if match else "N/A"
        except:
            return "N/A"

    def get_reverse_dns_safe(self, ip):
        """DNS safe pour proxychains"""
        try:
            host = socket.gethostbyaddr(ip)[0]
            return host if host != ip else "N/A"
        except:
            return "N/A"

    def get_mac_address(self, ip):
        """MAC désactivée en proxychains (local seulement)"""
        if self.proxychains_mode:
            return "N/A (proxy)"
        # Code MAC normal...
        return "N/A"

    def get_vendor(self, mac):
        return "N/A" if self.proxychains_mode else self._get_vendor(mac)

    def _get_vendor(self, mac):
        if not mac or mac == "N/A":
            return "N/A"
        oui_db = {'00:0C:29': 'VMware', '00:50:56': 'VMware', '00:10:18': 'Microsoft'}
        mac_upper = mac.upper().replace('-', ':')
        for prefix, vendor in oui_db.items():
            if mac_upper.startswith(prefix):
                return vendor
        return "Unknown"

    def save_result(self, result):
        """Sauvegarde ATOMIQUE"""
        try:
            timestamp = datetime.now().strftime("%H:%M:%S")
            with open("netbios_pro_results.txt", "a", encoding='utf-8') as f:
                f.write(f"[{timestamp}] IP: {result['IP']} | Port: {result['Port']}\n")
                f.write(f"  Hostname: {result['Hostname']} | NetBIOS: {result['NetBIOS']}\n")
                f.write(f"  MAC: {result['MAC']} | Vendor: {result['Vendor']}\n\n")
        except:
            pass

    def print_result(self, result):
        if not self.running:
            return
        ip = result['IP']
        port = result['Port']
        hostname = result['Hostname']
        netbios = result['NetBIOS']
        
        print(f"\n{Fore.GREEN}🎯 [{port}] {Fore.CYAN}{ip}{Fore.GREEN}")
        print(f"   Host: {Fore.YELLOW}{hostname} | NB: {Fore.MAGENTA}{netbios}{Style.RESET_ALL}")
        self.save_result(result)

    def scan_single_ip(self, ip):
        if not self.running:
            return
        try:
            port = self.scan_port_proxychains(ip)
            if port:
                result = {
                    'IP': ip,
                    'Port': port,
                    'Hostname': self.get_reverse_dns_safe(ip),
                    'NetBIOS': self.get_netbios_proxy_safe(ip),
                    'MAC': self.get_mac_address(ip),
                    'Vendor': self.get_vendor(self.get_mac_address(ip))
                }
                
                with self.lock:
                    self.found_ips += 1
                    self.results.append(result)
                
                self.print_result(result)
            
            with self.lock:
                self.scanned_ips += 1
                
        except:
            with self.lock:
                self.scanned_ips += 1

    def display_progress(self):
        """Progression simplifiée non-bloquante"""
        while self.running:
            with self.lock:
                percent = (self.scanned_ips / self.total_ips * 100) if self.total_ips else 0
                elapsed = time.time() - self.start_time
                speed = self.scanned_ips / elapsed if elapsed > 0 else 0

            bar_len = 30
            filled = int(bar_len * min(percent / 100, 1))
            bar = f"{Fore.GREEN}█" * filled + f"{Fore.RED}░" * (bar_len - filled)

            self.safe_print(f'\r{Fore.CYAN}[{bar}] {percent:4.0f}% | '
                          f'{self.scanned_ips:>4}/{self.total_ips} | '
                          f'{Fore.GREEN}🎯{self.found_ips}{Style.RESET_ALL}')
            time.sleep(0.5)

    def load_ips(self, filename):
        ip_list = []
        try:
            with open(filename, 'r') as f:
                for line in f:
                    line = line.strip()
                    if line and not line.startswith('#'):
                        try:
                            if '/' in line:
                                network = ipaddress.ip_network(line, strict=False)
                                ip_list.extend([str(ip) for ip in network.hosts()])
                            else:
                                ipaddress.ip_address(line)
                                ip_list.append(line)
                        except:
                            continue
        except FileNotFoundError:
            print(f"{Fore.RED}❌ {filename} introuvable{Style.RESET_ALL}")
            sys.exit(1)
        self.total_ips = len(ip_list)
        return ip_list

    def save_all_results(self):
        if not self.results:
            return
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"netbios_proxychains_{timestamp}.txt"
        try:
            with open(filename, "w", encoding='utf-8') as f:
                f.write(f"PROXYCHAINS SCAN - {datetime.now()}\n")
                f.write(f"Total: {self.total_ips} | Scanné: {self.scanned_ips} | Trouvé: {self.found_ips}\n\n")
                for result in self.results:
                    f.write(f"IP: {result['IP']} | Port: {result['Port']} | "
                           f"Host: {result['Hostname']} | NetBIOS: {result['NetBIOS']}\n")
            print(f"{Fore.GREEN}💾 Backup: {filename}{Style.RESET_ALL}")
        except:
            pass

    def run(self, filename):
        self.print_banner()
        ip_list = self.load_ips(filename)
        
        print(f"{Fore.CYAN}📡 {len(ip_list):,} cibles | "
              f"Threads: 100 | Timeout: 1.2s{Style.RESET_ALL}")
        print(f"{Fore.YELLOW}Ctrl+C anytime → Sauvegarde garantie !")
        print()
        
        self.start_time = time.time()
        self.progress_thread = threading.Thread(target=self.display_progress, daemon=True)
        self.progress_thread.start()
        
        # Moins de threads pour proxychains (évite saturation)
        max_workers = 50 if self.proxychains_mode else 200
        
        self.executor = ThreadPoolExecutor(max_workers=max_workers)
        self.futures = [self.executor.submit(self.scan_single_ip, ip) for ip in ip_list]
        
        try:
            for future in as_completed(self.futures):
                if not self.running:
                    break
        except KeyboardInterrupt:
            pass
        
        self.running = False
        time.sleep(1)
        self.print_final_stats()

    def print_final_stats(self):
        elapsed = time.time() - self.start_time
        print(f"\n{Fore.GREEN}{'='*50}")
        print(f"🎉 TERMINÉ | {self.found_ips} targets")
        print(f"Scanné: {self.scanned_ips}/{self.total_ips}")
        print(f"Temps: {elapsed:.1f}s")
        print(f"{Fore.MAGENTA}Résultats → netbios_pro_results.txt{Style.RESET_ALL}")
        print(f"{Fore.GREEN}{'='*50}")

def main():
    if len(sys.argv) != 2:
        print(f"{Fore.RED}python3 {sys.argv[0]} ips.txt{Style.RESET_ALL}")
        return
    scanner = NetbiosProScanner()
    scanner.run(sys.argv[1])

if __name__ == "__main__":
    main()
